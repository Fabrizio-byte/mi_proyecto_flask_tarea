# Ejemplo de ruta para guardar datos de un formulario
@app.route('/guardar', methods=['POST'])
def guardar_datos_txt():
    # Asume que el formulario env√≠a un campo 'dato'
    dato = request.form['dato']
    with open('datos/datos.txt', 'a') as f:
        f.write(dato + '\n')
    return redirect(url_for('index'))
